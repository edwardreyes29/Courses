S.5.2 Partial Application
    -> taking one of the arguments and fixing the value.
    
    function add(x, y) {
        return x + y
    }

    function partiallyApply(func, x) {
        return function(y) {
            return func(x, y)
        }
    }

    e.g.
        var add5 = partiallyApply(add, 5)
        add5(2) // => 7
